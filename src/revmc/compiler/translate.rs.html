125    /// Translates an EVM bytecode into a native function.
126    ///
127    /// Example pseudo-code:
128    ///
129    /// ```ignore (pseudo-code)
130    /// // `cfg(may_suspend) = bytecode.may_suspend()`: `true` if it contains a
131    /// // `*CALL*` or `CREATE*` instruction.
132    /// fn evm_bytecode(args: ...) {
133    ///     setup_locals();
134    ///
135    ///     #[cfg(debug_assertions)]
136    ///     if args.<ptr>.is_null() { panic!("...") };
137    ///
138    ///     load_arguments();
139    ///
140    ///     #[cfg(may_suspend)]
141    ///     resume: {
142    ///         goto match ecx.resume_at {
143    ///             0 => inst0,
144    ///             1 => first_call_or_create_inst + 1, // + 1 as in the block after.
145    ///             2 => second_call_or_create_inst + 1,
146    ///             ... => ...,
147    ///             _ => unreachable, // Assumed to be valid.
148    ///         };
149    ///     };
150    ///     
151    ///     op.inst0: { /* ... */ };
152    ///     op.inst1: { /* ... */ };
153    ///     // ...
154    ///     #[cfg(may_suspend)]
155    ///     first_call_or_create_inst: {
156    ///          // ...
157    ///          goto suspend(1);
158    ///     };
159    ///     // ...
160    ///     // There will always be at least one diverging instruction.
161    ///     op.stop: {
162    ///         goto return(InstructionResult::Stop);
163    ///     };
164    ///
165    ///     #[cfg(may_suspend)]
166    ///     suspend(resume_at: u32): {
167    ///         ecx.resume_at = resume_at;
168    ///         goto return(InstructionResult::Stop);  // Caller checks next_action
169    ///     };
170    ///
171    ///     // All paths lead to here.
172    ///     return(ir: InstructionResult): {
173    ///         #[cfg(inspect_stack_length)]
174    ///         *args.stack_len = stack_len;
175    ///         return ir;
176    ///     }
177    /// }
178    /// ```
179    #[allow(rustdoc::invalid_rust_codeblocks)] // Syntax highlighting.
180    pub(super) fn translate(
181        mut bcx: B::Builder<'a>,
182        config: FcxConfig,
183        builtins: &'a mut Builtins<B>,
184        bytecode: &'a Bytecode<'a>,
185    ) -> Result<()> {
186        let entry_block = bcx.current_block().unwrap();
187
188        // Get common types.
189        let ptr_type = bcx.type_ptr();
190        let isize_type = bcx.type_ptr_sized_int();
191        let i8_type = bcx.type_int(8);
192        let i64_type = bcx.type_int(64);
193        let address_type = bcx.type_int(160);
194        let word_type = bcx.type_int(256);
195
196        // Set up entry block.
197        let gas_ptr = bcx.fn_param(0);
198        let gas_remaining = {
199            let offset = bcx.iconst(i64_type, mem::offset_of!(pf::Gas, remaining) as i64);
200            let name = "gas.remaining.addr";
201            Pointer::new_address(i64_type, bcx.gep(i8_type, gas_ptr, &[offset], name))
202        };
203
204        let sp_arg = bcx.fn_param(1);
205        let stack = if config.local_stack {
206            bcx.new_stack_slot(word_type, "stack.addr")
207        } else {
208            Pointer::new_address(word_type, sp_arg)
209        };
210
211        let stack_len_arg = bcx.fn_param(2);
212        // This is initialized later in `post_entry_block`.
213        let stack_len = bcx.new_stack_slot(isize_type, "len.addr");
214
215        let input = bcx.fn_param(3);
216        let ecx = bcx.fn_param(4);
217
218        // Create all instruction entry blocks.
219        let unreachable_block = bcx.create_block("unreachable");
220        let inst_entries: Vec<_> = bytecode
221            .iter_all_insts()
222            .map(|(i, data)| {
223                if data.is_dead_code() {
224                    unreachable_block
225                } else {
226                    bcx.create_block(&bytecode.op_block_name(i, ""))
227                }
228            })
229            .collect();
230        assert!(!inst_entries.is_empty(), "translating empty bytecode");
231
232        let dynamic_jump_table = bcx.create_block("dynamic_jump_table");
233        let suspend_block = bcx.create_block("suspend");
234        let failure_block = bcx.create_block("failure");
235        let return_block = bcx.create_block("return");
236
237        let mut fx = FunctionCx {
238            config,
239
240            ptr_type,
241            isize_type,
242            address_type,
243            word_type,
244            i8_type,
245            stack_len,
246            stack,
247            gas_remaining,
248            input,
249            ecx,
250            len_before: bcx.iconst(isize_type, 0),
251            len_offset: 0,
252            bcx,
253
254            bytecode,
255            inst_entries,
256            current_inst: usize::MAX,
257
258            incoming_dynamic_jumps: Vec::new(),
259            dynamic_jump_table,
260
261            incoming_failures: Vec::new(),
262            failure_block: Some(failure_block),
263            incoming_returns: Vec::new(),
264            return_block: Some(return_block),
265
266            resume_kind: ResumeKind::Indexes,
267            resume_blocks: Vec::new(),
268            suspend_blocks: Vec::new(),
269            suspend_block,
270
271            builtins,
272        };
273
274        // We store the stack length if requested or necessary due to the bytecode.
275        let stack_length_observable = config.inspect_stack_length || bytecode.may_suspend();
276
277        // Add debug assertions for the parameters.
278        if config.debug_assertions {
279            fx.pointer_panic_with_bool(
280                config.gas_metering,
281                gas_ptr,
282                "gas pointer",
283                "gas metering is enabled",
284            );
285            fx.pointer_panic_with_bool(
286                !config.local_stack,
287                sp_arg,
288                "stack pointer",
289                "local stack is disabled",
290            );
291            fx.pointer_panic_with_bool(
292                stack_length_observable,
293                stack_len_arg,
294                "stack length pointer",
295                if config.inspect_stack_length {
296                    "stack length inspection is enabled"
297                } else {
298                    "bytecode suspends execution"
299                },
300            );
301            fx.pointer_panic_with_bool(true, input, "input pointer", "");
302            fx.pointer_panic_with_bool(true, ecx, "EVM context pointer", "");
303        }
304
305        // The bytecode is guaranteed to have at least one instruction.
306        let first_inst_block = fx.inst_entries[0];
307        let post_entry_block = fx.bcx.create_block_after(entry_block, "entry.post");
308        let resume_block = fx.bcx.create_block_after(post_entry_block, "resume");
309        fx.bcx.br(post_entry_block);
310
311        // Translate individual instructions into their respective blocks.
312        for (inst, _) in bytecode.iter_insts() {
313            fx.translate_inst(inst)?;
314        }
315
316        // Finalize the dynamic jump table.
317        fx.bcx.switch_to_block(unreachable_block);
318        fx.bcx.unreachable();
319        if bytecode.has_dynamic_jumps() {
320            fx.bcx.switch_to_block(fx.dynamic_jump_table);
321            // TODO: Manually reduce to i32?
322            let jumpdests = bytecode.iter_insts().filter(|(_, data)| data.opcode == op::JUMPDEST);
323            // let max_pc =
324            //     jumpdests.clone().map(|(_, data)| data.pc).next_back().expect("no jumpdests");
325            let targets = jumpdests
326                .map(|(inst, data)| (data.pc as u64, fx.inst_entries[inst]))
327                .collect::<Vec<_>>();
328            let index = fx.bcx.phi(fx.word_type, &fx.incoming_dynamic_jumps);
329            // let target =
330            //     fx.bcx.create_block_after(fx.dynamic_jump_table, "dynamic_jump_table.contd");
331            // let overflow = fx.bcx.icmp_imm(IntCC::UnsignedGreaterThan, index, max_pc as i64);
332            // fx.bcx.brif(overflow, default, target);
333
334            // fx.bcx.switch_to_block(target);
335            // let index = fx.bcx.ireduce(i32_type, index);
336            fx.add_invalid_jump();
337            fx.bcx.switch(index, return_block, &targets, true);
338        } else {
339            // No dynamic jumps.
340            debug_assert!(fx.incoming_dynamic_jumps.is_empty());
341            fx.bcx.switch_to_block(fx.dynamic_jump_table);
342            fx.bcx.unreachable();
343        }
344
345        // Finalize the suspend and resume blocks. Must come before the return block.
346        // Also here is where the stack length is initialized.
347        let load_len_at_start = |fx: &mut Self| {
348            // Loaded from args only for the config.
349            if config.inspect_stack_length {
350                let stack_len = fx.bcx.load(fx.isize_type, stack_len_arg, "stack_len");
351                fx.stack_len.store(&mut fx.bcx, stack_len);
352            } else {
353                fx.stack_len.store_imm(&mut fx.bcx, 0);
354            }
355        };
356        let generate_resume = bytecode.may_suspend();
357        if generate_resume {
358            let get_ecx_resume_at_ptr = |fx: &mut Self| {
359                fx.get_field(
360                    fx.ecx,
361                    mem::offset_of!(EvmContext<'_>, resume_at),
362                    "ecx.resume_at.addr",
363                )
364            };
365
366            let kind = fx.resume_kind;
367            let resume_ty = match kind {
368                ResumeKind::Blocks => fx.ptr_type,
369                ResumeKind::Indexes => fx.isize_type,
370            };
371
372            // Resume block: load the `resume_at` value and switch to the corresponding block.
373            // Invalid values are treated as unreachable.
374            {
375                // Special-case the no resume case to load 0 into the length if possible.
376                let no_resume_block = fx.bcx.create_block_after(resume_block, "no_resume");
377
378                fx.bcx.switch_to_block(post_entry_block);
379                let resume_at = get_ecx_resume_at_ptr(&mut fx);
380                let resume_at = fx.bcx.load(resume_ty, resume_at, "ecx.resume_at");
381                let no_resume = match kind {
382                    ResumeKind::Blocks => fx.bcx.is_null(resume_at),
383                    ResumeKind::Indexes => fx.bcx.icmp_imm(IntCC::Equal, resume_at, 0),
384                };
385                fx.bcx.brif(no_resume, no_resume_block, resume_block);
386
387                fx.bcx.switch_to_block(no_resume_block);
388                load_len_at_start(&mut fx);
389                fx.bcx.br(first_inst_block);
390
391                // Dispatch to the resume block.
392                fx.bcx.switch_to_block(resume_block);
393                let stack_len = fx.bcx.load(fx.isize_type, stack_len_arg, "stack_len");
394                fx.stack_len.store(&mut fx.bcx, stack_len);
395                match kind {
396                    ResumeKind::Blocks => {
397                        fx.bcx.br_indirect(resume_at, &fx.resume_blocks);
398                    }
399                    ResumeKind::Indexes => {
400                        let default = fx.bcx.create_block_after(resume_block, "resume_invalid");
401                        fx.bcx.switch_to_block(default);
402                        fx.call_panic("invalid `resume_at` value");
403
404                        fx.bcx.switch_to_block(resume_block);
405                        let targets = fx
406                            .resume_blocks
407                            .iter()
408                            .enumerate()
409                            .map(|(i, b)| (i as u64 + 1, *b))
410                            .collect::<Vec<_>>();
411                        fx.bcx.switch(resume_at, default, &targets, true);
412                    }
413                }
414            }
415
416            // Suspend block: store the `resume_at` value and return `Stop`.
417            {
418                fx.bcx.switch_to_block(fx.suspend_block);
419                let resume_value = fx.bcx.phi(resume_ty, &fx.suspend_blocks);
420                let resume_at = get_ecx_resume_at_ptr(&mut fx);
421                fx.bcx.store(resume_value, resume_at);
422
423                // Signal that execution suspended - caller checks next_action for Call/Create
424                fx.build_return_imm(InstructionResult::Stop);
425            }
426        } else {
427            debug_assert!(fx.resume_blocks.is_empty());
428            debug_assert!(fx.suspend_blocks.is_empty());
429
430            fx.bcx.switch_to_block(post_entry_block);
431            load_len_at_start(&mut fx);
432            fx.bcx.br(first_inst_block);
433
434            fx.bcx.switch_to_block(resume_block);
435            fx.bcx.unreachable();
436            fx.bcx.switch_to_block(fx.suspend_block);
437            fx.bcx.unreachable();
438        }
439
440        // Finalize the failure block.
441        fx.bcx.switch_to_block(fx.failure_block.unwrap());
442        if !fx.incoming_failures.is_empty() {
443            let failure_value = fx.bcx.phi(fx.i8_type, &fx.incoming_failures);
444            fx.bcx.set_current_block_cold();
445            fx.build_return(failure_value);
446        } else {
447            fx.bcx.unreachable();
448        }
449
450        // Finalize the return block.
451        fx.bcx.switch_to_block(fx.return_block.unwrap());
452        if !fx.incoming_returns.is_empty() {
453            let return_value = fx.bcx.phi(fx.i8_type, &fx.incoming_returns);
454            if stack_length_observable {
455                fx.save_stack_len();
456            }
457            fx.bcx.ret(&[return_value]);
458        } else {
459            fx.bcx.unreachable();
460        }
461
462        fx.bcx.seal_all_blocks();
463
464        Ok(())
465    }

467    #[instrument(level = "debug", skip_all, fields(inst = %self.bytecode.inst(inst).to_op()))]
468    fn translate_inst(&mut self, inst: Inst) -> Result<()> {
469        self.current_inst = inst;
470        let data = self.bytecode.inst(inst);
471        let opcode = data.opcode;
472        let entry_block = self.inst_entries[inst];
473        self.bcx.switch_to_block(entry_block);
474
475        // self.call_printf(format_printf!("{}\n", self.op_block_name("")), &[]);
476
477        let branch_to_next_opcode = |this: &mut Self| {
478            debug_assert!(
479                !this.bytecode.is_instr_diverging(inst),
480                "attempted to branch to next instruction in a diverging instruction: {data:?}",
481            );
482            if let Some(next) = this.inst_entries.get(inst + 1) {
483                this.bcx.br(*next);
484            }
485        };
486        // Currently a noop.
487        // let epilogue = |this: &mut Self| {
488        //     this.bcx.seal_block(entry_block);
489        // };
490
491        /// Makes sure to run cleanup code and return.
492        /// Use `no_branch` to skip the branch to the next opcode.
493        /// Use `build` to build the return instruction and skip the branch.
494        macro_rules! goto_return {
495            (no_branch $($comment:expr)?) => {
496                $(
497                    if self.config.comments {
498                        self.add_comment($comment);
499                    }
500                )?
501                // epilogue(self);
502                return Ok(());
503            };
504            (build $ret:expr) => {{
505                self.build_return_imm($ret);
506                goto_return!(no_branch);
507            }};
508            (fail $ret:expr) => {{
509                self.build_fail_imm($ret);
510                goto_return!(no_branch);
511            }};
512            ($($comment:expr)?) => {
513                branch_to_next_opcode(self);
514                goto_return!(no_branch $($comment)?);
515            };
516        }
517
518        // Assert that we already skipped the block.
519        debug_assert!(!data.flags.contains(InstFlags::DEAD_CODE));
520
521        #[cfg(test)]
522        if opcode == crate::TEST_SUSPEND {
523            self.suspend();
524            goto_return!(no_branch);
525        }
526
527        // Disabled instructions don't pay gas.
528        if data.flags.contains(InstFlags::DISABLED) {
529            goto_return!(fail InstructionResult::NotActivated);
530        }
531        if data.flags.contains(InstFlags::UNKNOWN) {
532            goto_return!(fail InstructionResult::OpcodeNotFound);
533        }
534
535        // Pay static gas for the current section.
536        self.gas_cost_imm(data.section.gas_cost as u64);
537
538        if data.flags.contains(InstFlags::SKIP_LOGIC) {
539            goto_return!("skipped");
540        }
541
542        // Reset the stack length offset for this instruction.
543        self.len_offset = 0;
544        self.len_before = self.stack_len.load(&mut self.bcx, "stack_len");
545
546        // Check stack length for the current section.
547        if self.config.stack_bound_checks {
548            let inp = data.section.inputs;
549            let diff = data.section.max_growth as i64;
550
551            if diff > revmc_context::EvmStack::CAPACITY as i64 {
552                goto_return!(fail InstructionResult::StackOverflow);
553            }
554
555            let underflow = |this: &mut Self| {
556                debug_assert!(inp > 0);
557                this.bcx.icmp_imm(IntCC::UnsignedLessThan, this.len_before, inp as i64)
558            };
559            let overflow = |this: &mut Self| {
560                debug_assert!(diff > 0 && diff <= STACK_CAP as i64);
561                this.bcx.icmp_imm(
562                    IntCC::UnsignedGreaterThan,
563                    this.len_before,
564                    STACK_CAP as i64 - diff,
565                )
566            };
567
568            let may_underflow = inp > 0;
569            let may_overflow = diff > 0;
570            if may_underflow && may_overflow {
571                let underflow = underflow(self);
572                let overflow = overflow(self);
573                let cond = self.bcx.bitor(underflow, overflow);
574                let ret = {
575                    let under =
576                        self.bcx.iconst(self.i8_type, InstructionResult::StackUnderflow as i64);
577                    let over =
578                        self.bcx.iconst(self.i8_type, InstructionResult::StackOverflow as i64);
579                    self.bcx.select(underflow, under, over)
580                };
581                let target = self.build_check_inner(true, cond, ret);
582                self.bcx.switch_to_block(target);
583            } else if may_underflow {
584                let cond = underflow(self);
585                self.build_check(cond, InstructionResult::StackUnderflow);
586            } else if may_overflow {
587                let cond = overflow(self);
588                self.build_check(cond, InstructionResult::StackOverflow);
589            }
590        }
591
592        // Update the stack length for this instruction.
593        {
594            let (inp, out) = data.stack_io();
595            let diff = out as i64 - inp as i64;
596            if diff != 0 {
597                let mut diff = diff;
598                // HACK: For now all opcodes that suspend (minus the test one, which does not reach
599                // here) return exactly one value. This value is pushed onto the stack by the
600                // caller, so we don't account for it here.
601                if data.may_suspend() {
602                    diff -= 1;
603                }
604                let len_changed = self.bcx.iadd_imm(self.len_before, diff);
605                self.stack_len.store(&mut self.bcx, len_changed);
606            }
607        }
608
609        // Macro utils.
610        macro_rules! unop {
611            ($op:ident) => {{
612                let mut a = self.pop();
613                a = self.bcx.$op(a);
614                self.push(a);
615            }};
616        }
617
618        macro_rules! binop {
619            ($op:ident) => {{
620                let [a, b] = self.popn();
621                let r = self.bcx.$op(a, b);
622                self.push(r);
623            }};
624            (@shift $op:ident, | $value:ident, $shift:ident | $default:expr) => {{
625                let [$shift, $value] = self.popn();
626                let r = self.bcx.$op($value, $shift);
627                let overflow = self.bcx.icmp_imm(IntCC::UnsignedGreaterThan, $shift, 255);
628                let default = $default;

630                self.push(r);
631            }};
632            (@if_not_zero $op:ident) => {{
633                // Use `lazy_select` (conditional branch) instead of `select` to avoid
634                // computing `$op(a, b)` when `b == 0`. Division/remainder by zero is UB
635                // in LLVM IR, and `select` evaluates both operands unconditionally, so
636                // LLVM may exploit the UB to remove the zero check entirely.
637                let [a, b] = self.popn();
638                let b_is_zero = self.bcx.icmp_imm(IntCC::Equal, b, 0);
639                let r = self.bcx.lazy_select(
640                    b_is_zero,
641                    self.word_type,
642                    |bcx| bcx.iconst_256(U256::ZERO),
643                    |bcx| bcx.$op(a, b),
644                );
645                self.push(r);
646            }};
647        }

1068    /// Pushes 256-bit values onto the stack.
1069    fn pushn(&mut self, values: &[B::Value]) {
1070        let len_start = self.len_before();
1071        for &value in values {
1072            let len = if self.len_offset != 0 {
1073                self.bcx.iadd_imm(len_start, self.len_offset as i64)
1074            } else {
1075                len_start
1076            };
1077            self.len_offset += 1;
1078            let sp = self.sp_at(len);
1079            self.bcx.store(value, sp);
1080        }
1081    }

1088    /// Removes the topmost `N` elements from the stack and returns them.
1089    ///
1090    /// If `load` is `false`, returns just the pointers.
1091    fn popn<const N: usize>(&mut self) -> [B::Value; N] {
1092        debug_assert_ne!(N, 0);
1093
1094        let len_start = self.len_before();
1095        std::array::from_fn(|i| {
1096            self.len_offset -= 1;
1097            let len = if self.len_offset != 0 {
1098                self.bcx.iadd_imm(len_start, self.len_offset as i64)
1099            } else {
1100                len_start
1101            };
1102            let sp = self.sp_at(len);
1103            let name = b'a' + i as u8;
1104            self.load_word(sp, std::str::from_utf8(&[name]).unwrap())
1105        })
1106    }

1124    /// Exchange two values on the stack.
1125    /// `n` is the first index, and the second index is calculated as `n + m`.
1126    /// `m` cannot be `0`.
1127    fn exchange(&mut self, n: usize, m: usize) {
1128        debug_assert_ne!(m, 0);
1129        let len = self.len_before();
1130        // Load a.
1131        let a_sp = self.sp_from_top(len, n + 1);
1132        let a = self.load_word(a_sp, "swap.a");
1133        // Load b.
1134        let b_sp = self.sp_from_top(len, n + m + 1);
1135        let b = self.load_word(b_sp, "swap.b");
1136        // Store.
1137        self.bcx.store(a, b_sp);
1138        self.bcx.store(b, a_sp);
1139    }

1167    /// Suspend execution, storing the resume point in the context.
1168    fn suspend(&mut self) {
1169        // Register the next instruction as the resume block.
1170        let idx = self.resume_blocks.len();
1171        let value = self.add_resume_at(self.inst_entries[self.current_inst + 1]);
1172
1173        // Register the current block as the suspend block.
1174        let value = match value {
1175            Some(value) => value,
1176            None => self.bcx.iconst(self.isize_type, idx as i64 + 1),
1177        };
1178        self.suspend_blocks.push((value, self.bcx.current_block().unwrap()));
1179
1180        // Branch to the suspend block.
1181        self.bcx.br(self.suspend_block);
1182    }

1331    #[must_use]
1332    fn build_check_inner(
1333        &mut self,
1334        is_failure: bool,
1335        cond: B::Value,
1336        ret: B::Value,
1337    ) -> B::BasicBlock {
1338        let current_block = self.current_block();
1339        let target = self.create_block_after(current_block, "contd");
1340
1341        let return_block = if let Some(return_block) = self.return_block {
1342            self.incoming_returns.push((ret, current_block));
1343            return_block
1344        } else {
1345            self.create_block_after(target, "return")
1346        };
1347        let then_block = if is_failure { return_block } else { target };
1348        let else_block = if is_failure { target } else { return_block };
1349        self.bcx.brif_cold(cond, then_block, else_block, is_failure);
1350
1351        if self.return_block.is_none() {
1352            self.bcx.switch_to_block(return_block);
1353            self.bcx.ret(&[ret]);
1354        }

1527    /// Builds: `fn byte(index: u256, value: u256) -> u256`
1528    fn build_byte(&mut self) {
1529        let index = self.bcx.fn_param(0);
1530        let value = self.bcx.fn_param(1);
1531
1532        let cond = self.bcx.icmp_imm(IntCC::UnsignedLessThan, index, 32);
1533        let byte = {
1534            // (value >> (31 - index) * 8) & 0xFF
1535            let thirty_one = self.bcx.iconst_256(U256::from(31));
1536            let shift = self.bcx.isub(thirty_one, index);
1537            let shift = self.bcx.imul_imm(shift, 8);
1538            let shifted = self.bcx.ushr(value, shift);
1539            let mask = self.bcx.iconst_256(U256::from(0xFF));
1540            self.bcx.bitand(shifted, mask)
1541        };
1542        let zero = self.bcx.iconst_256(U256::ZERO);
1543        let r = self.bcx.select(cond, byte, zero);
1544
1545        self.bcx.ret(&[r]);
1546    }

1552    /// Builds: `fn signextend(ext: u256, x: u256) -> u256`
1553    fn build_signextend(&mut self) {
1554        // From the yellow paper:
1555        /*
1556        let [ext, x] = stack.pop();
1557        let t = 256 - 8 * (ext + 1);
1558        let mut result = x;
1559        result[..t] = [x[t]; t]; // Index by bits.
1560        */
1561
1562        let ext = self.bcx.fn_param(0);
1563        let x = self.bcx.fn_param(1);
1564
1565        // For 31 we also don't need to do anything.
1566        let might_do_something = self.bcx.icmp_imm(IntCC::UnsignedLessThan, ext, 31);
1567        let r = self.bcx.lazy_select(
1568            might_do_something,
1569            self.bcx.type_int(256),
1570            |bcx| {
1571                // Adapted from revm: https://github.com/bluealloy/revm/blob/fda371f73aba2c30a83c639608be78145fd1123b/crates/interpreter/src/instructions/arithmetic.rs#L89
1572                // let bit_index = 8 * ext + 7;
1573                // let bit = (x >> bit_index) & 1 != 0;
1574                // let mask = (1 << bit_index) - 1;
1575                // let r = if bit { x | !mask } else { *x & mask };
1576
1577                // let bit_index = 8 * ext + 7;
1578                let bit_index = bcx.imul_imm(ext, 8);
1579                let bit_index = bcx.iadd_imm(bit_index, 7);
1580
1581                // let bit = (x >> bit_index) & 1 != 0;
1582                let one = bcx.iconst_256(U256::from(1));
1583                let bit = bcx.ushr(x, bit_index);
1584                let bit = bcx.bitand(bit, one);
1585                let bit = bcx.icmp_imm(IntCC::NotEqual, bit, 0);
1586
1587                // let mask = (1 << bit_index) - 1;
1588                let mask = bcx.ishl(one, bit_index);
1589                let mask = bcx.isub_imm(mask, 1);
1590
1591                // let r = if bit { x | !mask } else { *x & mask };
1592                let not_mask = bcx.bitnot(mask);
1593                let sext = bcx.bitor(x, not_mask);
1594                let zext = bcx.bitand(x, mask);
1595                bcx.select(bit, sext, zext)
1596            },
1597            |_bcx| x,
1598        );
1599        self.bcx.ret(&[r]);
1600    }

1603        &mut self,
1604        name: &str,
1605        x1: B::Value,

1608    ) -> B::Value {

1613    #[must_use]
1614    fn call_ir_builtin(
1615        &mut self,
1616        name: &str,
1617        args: &[B::Value],
1618        arg_types: &[B::Type],
1619        ret: Option<B::Type>,
1620        build: impl FnOnce(&mut Self),
1621    ) -> Option<B::Value> {
1622        let prefix = "__revmc_ir_builtin_";
1623        let name = &format!("{prefix}{name}")[..];
1624
1625        // self.call_printf(format_printf!("{} - calling {name}\n", self.op_block_name("")), &[]);
1626
1627        debug_assert_eq!(args.len(), arg_types.len());
1628        let linkage = revmc_backend::Linkage::Private;
1629        let this = unsafe { std::mem::transmute::<&mut Self, &mut Self>(self) };
1630        let f = self.bcx.get_or_build_function(name, arg_types, ret, linkage, |bcx| {
1631            let prev_return_block = this.return_block.take();
1632            let prev_failure_block = this.failure_block.take();
1633            mem::swap(&mut this.bcx, bcx);
1634
1635            for attr in default_attrs::for_fn().chain(std::iter::once(Attribute::NoUnwind)) {
1636                this.bcx.add_function_attribute(None, attr, FunctionAttributeLocation::Function)
1637            }
1638            for i in 0..this.bcx.num_fn_params() as u32 {
1639                for attr in default_attrs::for_param() {
1640                    this.bcx.add_function_attribute(None, attr, FunctionAttributeLocation::Param(i))
1641                }
1642            }
1643            build(this);
1644
1645            mem::swap(&mut this.bcx, bcx);
1646            this.failure_block = prev_failure_block;
1647            this.return_block = prev_return_block;
1648        });
1649        self.bcx.call(f, args)
1650    }
