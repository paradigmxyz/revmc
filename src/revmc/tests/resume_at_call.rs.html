1// Tests for resume_at persistence in call_with_interpreter (49274dbb).
2//
3// When JIT code encounters a CALL instruction, it suspends execution and
4// returns `InterpreterAction::NewFrame`. After the callee completes, the
5// caller must resume from the instruction *after* the CALL — not from PC=0.
6//
7// Without the fix, `call_with_interpreter` did not persist `ecx.resume_at`
8// back into the interpreter's bytecode PC, causing re-execution from the
9// beginning on every re-entry.
10
11use super::{
12    insert_call_outcome_test, TestHost, DEF_ADDR, DEF_CALLER, DEF_CD, DEF_GAS_LIMIT, DEF_SPEC,
13    DEF_VALUE,
14};
15use crate::{Backend, EvmCompiler};
16use revm_bytecode::opcode as op;
17use revm_interpreter::{
18    interpreter::ExtBytecode, CallInput, FrameInput, Gas, InputsImpl, InstructionResult,
19    Interpreter, InterpreterAction, InterpreterResult, SharedMemory,
20};
21use revm_primitives::{Bytes, U256};
22
23matrix_tests!(call_then_push = |compiler| run_call_then_push(compiler));
24matrix_tests!(call_then_return = |compiler| run_call_then_return(compiler));
25
26/// Contract: PUSH args → CALL → PUSH1 0x42 → STOP
27///
28/// After the CALL returns, execution must continue with PUSH1 0x42 (not restart
29/// from PUSH args). We verify by checking the stack after the second call to
30/// `call_with_interpreter`.
31fn run_call_then_push<B: Backend>(compiler: &mut EvmCompiler<B>) {
32    #[rustfmt::skip]
33    let bytecode: &[u8] = &[
34        // Set up CALL arguments (7 stack items)
35        op::PUSH1, 0,    // ret length
36        op::PUSH1, 0,    // ret offset
37        op::PUSH1, 0,    // args length
38        op::PUSH1, 0,    // args offset
39        op::PUSH1, 0,    // value = 0
40        op::PUSH1, 0x69, // address
41        op::GAS,         // gas (all remaining)
42        op::CALL,        // suspends here with NewFrame
43        // -- after CALL returns, execution resumes here --
44        op::PUSH1, 0x42, // push marker value
45        op::STOP,
46    ];
47
48    unsafe { compiler.clear() }.unwrap();
49    compiler.inspect_stack_length(true);
50    let f = unsafe { compiler.jit("resume_call", bytecode, DEF_SPEC) }.unwrap();
51
52    // First call: should suspend at CALL with NewFrame
53    let mut host = TestHost::new();
54    let input = InputsImpl {
55        target_address: DEF_ADDR,
56        bytecode_address: None,
57        caller_address: DEF_CALLER,
58        input: CallInput::Bytes(Bytes::from_static(DEF_CD)),
59        call_value: DEF_VALUE,
60    };
61    let bytecode_obj = revm_bytecode::Bytecode::new_raw(Bytes::copy_from_slice(bytecode));
62    let ext_bytecode = ExtBytecode::new(bytecode_obj);
63    let mut interpreter =
64        Interpreter::new(SharedMemory::new(), ext_bytecode, input, false, DEF_SPEC, DEF_GAS_LIMIT);
65
66    let action = unsafe { f.call_with_interpreter(&mut interpreter, &mut host) };
67
68    // Should get NewFrame(Call(...))
69    let return_memory_offset = match &action {
70        InterpreterAction::NewFrame(FrameInput::Call(call_inputs)) => {
71            Some(call_inputs.return_memory_offset.clone())
72        }
73        other => panic!("expected NewFrame(Call), got {other:?}"),
74    };
75
76    // Simulate the callee completing successfully with no output
77    let call_result = InterpreterResult {
78        result: InstructionResult::Stop,
79        output: Bytes::new(),
80        gas: Gas::new(0),
81    };
82    insert_call_outcome_test(&mut interpreter, call_result, return_memory_offset);
83
84    // Second call: should resume after CALL, execute PUSH1 0x42, STOP
85    let action = unsafe { f.call_with_interpreter(&mut interpreter, &mut host) };
86
87    match &action {
88        InterpreterAction::Return(result) => {
89            assert_eq!(
90                result.result,
91                InstructionResult::Stop,
92                "expected Stop after resume, got {:?}",
93                result.result
94            );
95            // Stack should have: [call_success_indicator(1), 0x42]
96            // The CALL pushed success=1, then PUSH1 0x42
97            assert_eq!(interpreter.stack.len(), 2, "stack should have 2 items");
98            assert_eq!(
99                interpreter.stack.data()[1],
100                U256::from(0x42),
101                "top of stack should be 0x42 (the marker value pushed after CALL)"
102            );
103        }
104        other => panic!("expected Return after resume, got {other:?}"),
105    }
106}

108/// Contract: PUSH args → CALL → PUSH1 32 → PUSH0 → RETURN
109///
110/// After the CALL returns, execution should RETURN 32 zero bytes.
111/// Without the resume_at fix, it would re-enter from the beginning and try
112/// to CALL again, never reaching the RETURN.
113fn run_call_then_return<B: Backend>(compiler: &mut EvmCompiler<B>) {
114    #[rustfmt::skip]
115    let bytecode: &[u8] = &[
116        // CALL arguments
117        op::PUSH1, 0,    // ret length
118        op::PUSH1, 0,    // ret offset
119        op::PUSH1, 0,    // args length
120        op::PUSH1, 0,    // args offset
121        op::PUSH1, 0,    // value = 0
122        op::PUSH1, 0x69, // address
123        op::GAS,         // gas
124        op::CALL,        // suspends
125        // -- resume point --
126        op::POP,          // pop call success indicator
127        op::PUSH1, 32,   // return size
128        op::PUSH0,       // return offset
129        op::RETURN,       // should return 32 zero bytes
130    ];
131
132    unsafe { compiler.clear() }.unwrap();
133    compiler.inspect_stack_length(true);
134    let f = unsafe { compiler.jit("resume_return", bytecode, DEF_SPEC) }.unwrap();
135
136    let mut host = TestHost::new();
137    let input = InputsImpl {
138        target_address: DEF_ADDR,
139        bytecode_address: None,
140        caller_address: DEF_CALLER,
141        input: CallInput::Bytes(Bytes::from_static(DEF_CD)),
142        call_value: DEF_VALUE,
143    };
144    let bytecode_obj = revm_bytecode::Bytecode::new_raw(Bytes::copy_from_slice(bytecode));
145    let ext_bytecode = ExtBytecode::new(bytecode_obj);
146    let mut interpreter =
147        Interpreter::new(SharedMemory::new(), ext_bytecode, input, false, DEF_SPEC, DEF_GAS_LIMIT);
148
149    // First call: suspends at CALL
150    let action = unsafe { f.call_with_interpreter(&mut interpreter, &mut host) };
151    let return_memory_offset = match &action {
152        InterpreterAction::NewFrame(FrameInput::Call(call_inputs)) => {
153            Some(call_inputs.return_memory_offset.clone())
154        }
155        other => panic!("expected NewFrame(Call), got {other:?}"),
156    };
157
158    // Simulate callee returning
159    let call_result = InterpreterResult {
160        result: InstructionResult::Stop,
161        output: Bytes::new(),
162        gas: Gas::new(0),
163    };
164    insert_call_outcome_test(&mut interpreter, call_result, return_memory_offset);
165
166    // Second call: should resume, POP, PUSH1 32, PUSH0, RETURN
167    let action = unsafe { f.call_with_interpreter(&mut interpreter, &mut host) };
168
169    match &action {
170        InterpreterAction::Return(result) => {
171            assert_eq!(result.result, InstructionResult::Return);
172            assert_eq!(result.output.len(), 32, "expected 32-byte return output");
173        }
174        other => panic!("expected Return after resume, got {other:?}"),
175    }
176}
