72    (@cases $( $name:ident($($t:tt)*) ),* $(,)?) => {

78    (@case @raw { $($fields:tt)* }) => { &TestCase { $($fields)* ..Default::default() } };
79
80    (@case $op:expr $(, $args:expr)* $(,)? => $($ret:expr),* $(,)? $(; op_gas($op_gas:expr))?) => {
81        &TestCase {
82            bytecode: &tests!(@bytecode $op, $($args),*),
83            expected_stack: &[$($ret),*],

89    (@bytecode $op:expr, $a:expr) => { bytecode_unop($op, $a) };
