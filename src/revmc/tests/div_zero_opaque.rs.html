1//! Regression test: DIV/MOD/SMOD with zero divisor loaded from memory.
2//!
3//! When the divisor is opaque to LLVM (e.g. loaded via MLOAD), `udiv/urem/srem
4//! i256` with divisor==0 is UB in LLVM IR. The old `binop!(@if_not_zero ...)`
5//! used `select` which evaluates both operands unconditionally, allowing LLVM to
6//! exploit the UB and remove the zero check entirely. The fix uses `lazy_select`
7//! (conditional branch) so the division is never executed when b==0.
8
9use super::{with_evm_context, DEF_SPEC};
10use crate::{Backend, EvmCompiler};
11use revm_bytecode::opcode as op;
12use revm_interpreter::InstructionResult;
13use revm_primitives::U256;
14
15/// Build bytecode: MSTORE(a, 0), MSTORE(b, 32), MLOAD(32), MLOAD(0), `<op>`
16///
17/// The operands pass through MLOAD (an opaque builtin), preventing LLVM from
18/// constant-folding or exploiting division-by-zero UB at compile time.
19fn bytecode_binop_opaque(opcode: u8, a: U256, b: U256) -> Vec<u8> {
20    let mut code = Vec::with_capacity(128);
21
22    // Store a at memory[0]
23    code.push(op::PUSH32);
24    code.extend_from_slice(&a.to_be_bytes::<32>());
25    code.push(op::PUSH1);
26    code.push(0x00);
27    code.push(op::MSTORE);
28
29    // Store b at memory[32]
30    code.push(op::PUSH32);
31    code.extend_from_slice(&b.to_be_bytes::<32>());
32    code.push(op::PUSH1);
33    code.push(0x20);
34    code.push(op::MSTORE);
35
36    // Load b from memory (opaque to LLVM)
37    code.push(op::PUSH1);
38    code.push(0x20);
39    code.push(op::MLOAD);
40
41    // Load a from memory (opaque to LLVM)
42    code.push(op::PUSH1);
43    code.push(0x00);
44    code.push(op::MLOAD);
45
46    // Execute the target opcode: OP(a, b)
47    code.push(opcode);
48
49    code
50}
51
52fn run_opaque_zero_div_test<B: Backend>(
53    compiler: &mut EvmCompiler<B>,
54    name: &str,
55    opcode: u8,
56    a: U256,
57) {
58    let code = bytecode_binop_opaque(opcode, a, U256::ZERO);
59
60    unsafe { compiler.clear() }.unwrap();
61    compiler.inspect_stack_length(true);
62    let f = unsafe { compiler.jit(name, &code, DEF_SPEC) }.unwrap();
63
64    with_evm_context(&code, |ecx, stack, stack_len| {
65        let r = unsafe { f.call(Some(stack), Some(stack_len), ecx) };
66        assert_eq!(r, InstructionResult::Stop, "{name}: unexpected return");
67        assert_eq!(*stack_len, 1, "{name}: expected 1 stack element");
68        let actual = stack.as_slice()[0].to_u256();
69        assert_eq!(
70            actual,
71            U256::ZERO,
72            "{name}: JIT produced {actual} but EVM spec mandates 0 for division by zero"
73        );
74    });
75}
76
77// EVM spec: x DIV 0 = 0, x MOD 0 = 0, x SMOD 0 = 0, x SDIV 0 = 0
78matrix_tests!(div_zero = |jit| run_opaque_zero_div_test(jit, "div_zero", op::DIV, U256::from(32)));
79matrix_tests!(mod_zero = |jit| run_opaque_zero_div_test(jit, "mod_zero", op::MOD, U256::from(32)));
80matrix_tests!(
81    smod_zero = |jit| run_opaque_zero_div_test(jit, "smod_zero", op::SMOD, U256::from(5))
82);
