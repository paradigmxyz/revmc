24/// An EVM instruction is a high level internal representation of an EVM opcode.
25///
26/// This is an index into [`Bytecode`] instructions.
27///
28/// Also known as `ic`, or instruction counter; not to be confused with SSA `inst`s.
29pub(crate) type Inst = usize;
30
31/// EVM bytecode.
32#[doc(hidden)] // Not public API.
33pub struct Bytecode<'a> {
34    /// The original bytecode slice.
35    pub(crate) code: &'a [u8],
36    /// The instructions.
37    insts: Vec<InstData>,
38    /// `JUMPDEST` opcode map. `jumpdests[pc]` is `true` if `code[pc] == op::JUMPDEST`.
39    jumpdests: BitVec,
40    /// The [`SpecId`].
41    pub(crate) spec_id: SpecId,
42    /// Whether the bytecode contains dynamic jumps.
43    has_dynamic_jumps: bool,
44    /// Whether the bytecode may suspend execution.
45    may_suspend: bool,
46    /// Mapping from program counter to instruction.
47    pc_to_inst: FxHashMap<u32, u32>,
48}

50impl<'a> Bytecode<'a> {
51    #[instrument(name = "new_bytecode", level = "debug", skip_all)]
52    pub(crate) fn new(code: &'a [u8], spec_id: SpecId) -> Self {
53        let mut insts = Vec::with_capacity(code.len() + 8);
54        let mut jumpdests = BitVec::repeat(false, code.len());
55        let mut pc_to_inst = FxHashMap::with_capacity_and_hasher(code.len(), Default::default());
56        let op_infos = op_info_map(spec_id);
57        for (inst, (pc, Opcode { opcode, immediate: _ })) in
58            OpcodesIter::new(code, spec_id).with_pc().enumerate()
59        {
60            pc_to_inst.insert(pc as u32, inst as u32);
61
62            if opcode == op::JUMPDEST {
63                jumpdests.set(pc, true)
64            }
65
66            let data = 0;
67
68            let mut flags = InstFlags::empty();
69            let info = op_infos[opcode as usize];
70            if info.is_unknown() {
71                flags |= InstFlags::UNKNOWN;
72            }
73            if info.is_disabled() {
74                flags |= InstFlags::DISABLED;
75            }
76            let base_gas = info.base_gas();
77
78            let section = Section::default();
79
80            insts.push(InstData { opcode, flags, base_gas, data, pc: pc as u32, section });
81        }
82
83        let mut bytecode = Self {
84            code,
85            insts,
86            jumpdests,
87            spec_id,
88            has_dynamic_jumps: false,
89            may_suspend: false,
90            pc_to_inst,
91        };
92
93        // Pad code to ensure there is at least one diverging instruction.
94        if bytecode.insts.last().is_none_or(|last| !last.is_diverging()) {
95            bytecode.insts.push(InstData::new(op::STOP));
96        }
97
98        bytecode
99    }

163    /// Runs a list of analysis passes on the instructions.
164    #[instrument(level = "debug", skip_all)]
165    pub(crate) fn analyze(&mut self) -> Result<()> {
166        self.static_jump_analysis();
167        // NOTE: `mark_dead_code` must run after `static_jump_analysis` as it can mark
168        // unreachable `JUMPDEST`s as dead code.
169        self.mark_dead_code();
170
171        self.calc_may_suspend();
172
173        self.construct_sections();
174
175        Ok(())
176    }
177
178    /// Mark `PUSH<N>` followed by `JUMP[I]` as `STATIC_JUMP` and resolve the target.
179    #[instrument(name = "sj", level = "debug", skip_all)]
180    fn static_jump_analysis(&mut self) {
181        for jump_inst in 0..self.insts.len() {
182            let jump = &self.insts[jump_inst];
183            let Some(push_inst) = jump_inst.checked_sub(1) else {
184                if jump.is_legacy_jump() {
185                    trace!(jump_inst, target=?None::<()>, "found jump");
186                    self.has_dynamic_jumps = true;
187                }
188                continue;
189            };
190
191            let push = &self.insts[push_inst];
192            if !(push.is_push() && jump.is_legacy_jump()) {
193                if jump.is_legacy_jump() {
194                    trace!(jump_inst, target=?None::<()>, "found jump");
195                    self.has_dynamic_jumps = true;
196                }
197                continue;
198            }
199
200            let imm_opt = self.get_imm(push);
201            if push.opcode != op::PUSH0 && imm_opt.is_none() {
202                continue;
203            }
204            let imm = imm_opt.unwrap_or(&[]);
205            self.insts[jump_inst].flags |= InstFlags::STATIC_JUMP;
206
207            const USIZE_SIZE: usize = std::mem::size_of::<usize>();
208            if imm.len() > USIZE_SIZE {
209                trace!(jump_inst, "jump target too large");
210                self.insts[jump_inst].flags |= InstFlags::INVALID_JUMP;
211                continue;
212            }
213
214            let mut padded = [0; USIZE_SIZE];
215            padded[USIZE_SIZE - imm.len()..].copy_from_slice(imm);
216            let target_pc = usize::from_be_bytes(padded);
217            if !self.is_valid_jump(target_pc) {
218                trace!(jump_inst, target_pc, "invalid jump target");
219                self.insts[jump_inst].flags |= InstFlags::INVALID_JUMP;
220                continue;
221            }
222
223            self.insts[push_inst].flags |= InstFlags::SKIP_LOGIC;
224            let target = self.pc_to_inst(target_pc);
225
226            // Mark the `JUMPDEST` as reachable.
227            debug_assert_eq!(
228                self.insts[target],
229                op::JUMPDEST,
230                "is_valid_jump returned true for non-JUMPDEST: \
231                 jump_inst={jump_inst} target_pc={target_pc} target={target}",
232            );
233            self.insts[target].data = 1;
234
235            // Set the target on the `JUMP` instruction.
236            trace!(jump_inst, target, "found jump");
237            self.insts[jump_inst].data = target as u32;
238        }
239    }

241    /// Mark unreachable instructions as `DEAD_CODE` to not generate any code for them.
242    ///
243    /// This pass is technically unnecessary as the backend will very likely optimize any
244    /// unreachable code that we generate, but this is trivial for us to do and significantly speeds
245    /// up code generation.
246    ///
247    /// We can simply mark all instructions that are between diverging instructions and
248    /// `JUMPDEST`s.
249    #[instrument(name = "dce", level = "debug", skip_all)]
250    fn mark_dead_code(&mut self) {
251        let mut iter = self.insts.iter_mut().enumerate();
252        while let Some((i, data)) = iter.next() {
253            if data.is_diverging() {
254                let mut end = i;
255                for (j, data) in &mut iter {
256                    end = j;
257                    if data.is_reachable_jumpdest(self.has_dynamic_jumps) {
258                        break;
259                    }
260                    data.flags |= InstFlags::DEAD_CODE;
261                }
262                let start = i + 1;
263                if end > start {
264                    debug!("found dead code: {start}..{end}");
265                }
266            }
267        }
268    }

270    /// Calculates whether the bytecode suspend suspend execution.
271    ///
272    /// This can only happen if the bytecode contains `*CALL*` or `*CREATE*` instructions.
273    #[instrument(name = "suspend", level = "debug", skip_all)]
274    fn calc_may_suspend(&mut self) {
275        let may_suspend = self.iter_insts().any(|(_, data)| data.may_suspend());
276        self.may_suspend = may_suspend;
277    }

279    /// Constructs the sections in the bytecode.
280    #[instrument(name = "sections", level = "debug", skip_all)]
281    fn construct_sections(&mut self) {
282        let mut analysis = SectionAnalysis::default();
283        for inst in 0..self.insts.len() {
284            if !self.inst(inst).is_dead_code() {
285                analysis.process(self, inst);
286            }
287        }
288        analysis.finish(self);
289    }
290
291    /// Constructs the sections in the bytecode.
292    #[instrument(name = "sections", level = "debug", skip_all)]
293    #[cfg(any())]
294    fn construct_sections_default(&mut self) {
295        for inst in &mut self.insts {
296            let (inp, out) = inst.stack_io();
297            let stack_diff = out as i16 - inp as i16;
298            inst.section =
299                Section { gas_cost: inst.base_gas as _, inputs: inp as _, max_growth: stack_diff }
300        }
301    }

303    /// Returns the immediate value of the given instruction data, if any.
304    /// Returns `None` if out of bounds too.
305    pub(crate) fn get_imm(&self, data: &InstData) -> Option<&'a [u8]> {
306        let imm_len = data.imm_len() as usize;
307        if imm_len == 0 {
308            return None;
309        }
310        let start = data.pc as usize + 1;
311        self.code.get(start..start + imm_len)
312    }

341    /// Converts a program counter (`self.code[pc]`) to an instruction (`self.inst(inst)`).
342    #[inline]

344        match self.pc_to_inst.get(&(pc as u32)) {

350    /// Returns the name for a basic block.
351    pub(crate) fn op_block_name(&self, inst: usize, name: &str) -> String {
352        use std::fmt::Write;
353
354        if inst == usize::MAX {
355            return format!("entry.{name}");
356        }
357        let data = self.inst(inst);
358
359        let mut s = String::new();
360        let _ = write!(s, "OP{inst}.{}", data.to_op());
361        if !name.is_empty() {
362            let _ = write!(s, ".{name}");
363        }
364        s
365    }

368impl fmt::Display for Bytecode<'_> {
369    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
370        let header = format!("{:^6} | {:^6} | {:^80} | {}", "ic", "pc", "opcode", "instruction");
371        writeln!(f, "{header}")?;
372        writeln!(f, "{}", "-".repeat(header.len()))?;
373        for (inst, (pc, opcode)) in self.opcodes().with_pc().enumerate() {
374            let data = self.inst(inst);
375            let opcode = opcode.to_string();
376            writeln!(f, "{inst:>6} | {pc:>6} | {opcode:<80} | {data:?}")?;
377        }
378        Ok(())
379    }
380}

395/// A single instruction in the bytecode.
396#[derive(Clone, Default)]
397pub(crate) struct InstData {
398    /// The opcode byte.
399    pub(crate) opcode: u8,
400    /// Flags.
401    pub(crate) flags: InstFlags,
402    /// The base gas cost of the opcode.
403    ///
404    /// This may not be the final/full gas cost of the opcode as it may also have a dynamic cost.
405    base_gas: u16,
406    /// Instruction-specific data:
407    /// - if the instruction has immediate data, this is a packed offset+length into the bytecode;
408    /// - `JUMP{,I} && STATIC_JUMP in kind`: the jump target, `Instr`;
409    /// - `JUMPDEST`: `1` if the jump destination is reachable, `0` otherwise;
410    /// - otherwise: no meaning.
411    pub(crate) data: u32,
412    /// The program counter, meaning `code[pc]` is this instruction's opcode.
413    pub(crate) pc: u32,
414    /// The section this instruction belongs to.
415    pub(crate) section: Section,
416}
417
418impl PartialEq<u8> for InstData {
419    #[inline]
420    fn eq(&self, other: &u8) -> bool {
421        self.opcode == *other
422    }
423}
424
425impl PartialEq<InstData> for u8 {
426    #[inline]
427    fn eq(&self, other: &InstData) -> bool {
428        *self == other.opcode
429    }
430}

432impl fmt::Debug for InstData {
433    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
434        f.debug_struct("InstData")
435            .field("opcode", &self.to_op())
436            .field("flags", &format_args!("{:?}", self.flags))
437            .field("data", &self.data)
438            .field("pc", &self.pc)
439            .field("section", &self.section)
440            .finish()
441    }

444impl InstData {
445    /// Creates a new instruction data with the given opcode byte.
446    /// Note that this may not be a valid instruction.
447    #[inline]
448    fn new(opcode: u8) -> Self {
449        Self { opcode, ..Default::default() }
450    }
451
452    /// Returns the length of the immediate data of this instruction.
453    #[inline]
454    pub(crate) const fn imm_len(&self) -> u8 {
455        min_imm_len(self.opcode)
456    }
457
458    /// Returns the number of input and output stack elements of this instruction.
459    #[inline]
460    pub(crate) fn stack_io(&self) -> (u8, u8) {
461        let (mut inp, out) = stack_io(self.opcode);
462        if self.is_legacy_static_jump()
463            && !(self.opcode == op::JUMPI && self.flags.contains(InstFlags::INVALID_JUMP))
464        {
465            inp -= 1;
466        }
467        (inp, out)
468    }
469
470    /// Converts this instruction to a raw opcode. Note that the immediate data is not resolved.
471    #[inline]
472    pub(crate) const fn to_op(&self) -> Opcode<'static> {
473        Opcode { opcode: self.opcode, immediate: None }
474    }
475
476    /// Converts this instruction to a raw opcode in the given bytecode.
477    #[inline]
478    #[allow(dead_code)]
479    pub(crate) fn to_op_in<'a>(&self, bytecode: &Bytecode<'a>) -> Opcode<'a> {
480        Opcode { opcode: self.opcode, immediate: bytecode.get_imm(self) }
481    }
482
483    /// Returns `true` if this instruction is a push instruction.
484    #[inline]
485    pub(crate) fn is_push(&self) -> bool {
486        matches!(self.opcode, op::PUSH0..=op::PUSH32)
487    }
488
489    /// Returns `true` if this instruction is a legacy jump instruction (`JUMP`/`JUMPI`).
490    #[inline]
491    pub(crate) fn is_legacy_jump(&self) -> bool {
492        matches!(self.opcode, op::JUMP | op::JUMPI)
493    }
494
495    /// Returns `true` if this instruction is a legacy jump instruction (`JUMP`/`JUMPI`), and the
496    /// target known statically.
497    #[inline]
498    pub(crate) fn is_legacy_static_jump(&self) -> bool {
499        self.is_legacy_jump() && self.flags.contains(InstFlags::STATIC_JUMP)
500    }
501
502    /// Returns `true` if this instruction is a `JUMPDEST`.
503    #[inline]
504    pub(crate) const fn is_jumpdest(&self) -> bool {
505        self.opcode == op::JUMPDEST
506    }
507
508    /// Returns `true` if this instruction is a reachable `JUMPDEST`.
509    #[inline]
510    pub(crate) const fn is_reachable_jumpdest(&self, has_dynamic_jumps: bool) -> bool {
511        self.is_jumpdest() && (has_dynamic_jumps || self.data == 1)
512    }
513
514    /// Returns `true` if this instruction is dead code.
515    pub(crate) fn is_dead_code(&self) -> bool {
516        self.flags.contains(InstFlags::DEAD_CODE)
517    }
518
519    /// Returns `true` if this instruction requires to know `gasleft()`.
520    /// Note that this does not include CALL and CREATE.
521    #[inline]
522    pub(crate) fn requires_gasleft(&self, spec_id: SpecId) -> bool {
523        // For SSTORE, see `revm_interpreter::gas::sstore_cost`.
524        self.opcode == op::GAS
525            || (self.opcode == op::SSTORE && spec_id.is_enabled_in(SpecId::ISTANBUL))
526    }
527
528    /// Returns `true` if we know that this instruction will branch or stop execution.
529    #[inline]
530    pub(crate) fn is_branching(&self) -> bool {
531        self.is_legacy_jump() || self.is_diverging()
532    }
533
534    /// Returns `true` if we know that this instruction will stop execution.
535    #[inline]
536    pub(crate) fn is_diverging(&self) -> bool {
537        #[cfg(test)]
538        if self.opcode == TEST_SUSPEND {
539            return false;
540        }
541
542        (self.opcode == op::JUMP && self.flags.contains(InstFlags::INVALID_JUMP))
543            || self.flags.contains(InstFlags::DISABLED)
544            || self.flags.contains(InstFlags::UNKNOWN)
545            || matches!(
546                self.opcode,
547                op::STOP | op::RETURN | op::REVERT | op::INVALID | op::SELFDESTRUCT
548            )
549    }
550
551    /// Returns `true` if this instruction may suspend execution.
552    #[inline]
553    pub(crate) const fn may_suspend(&self) -> bool {
554        #[cfg(test)]
555        if self.opcode == TEST_SUSPEND {
556            return true;
557        }
558
559        matches!(
560            self.opcode,
561            op::CALL | op::CALLCODE | op::DELEGATECALL | op::STATICCALL | op::CREATE | op::CREATE2
562        )
563    }
564}

566bitflags::bitflags! {
567    /// [`InstrData`] flags.
568    #[derive(Clone, Copy, Debug, Default, PartialEq, Eq)]
569    pub(crate) struct InstFlags: u8 {
570        /// The `JUMP`/`JUMPI` target is known at compile time.
571        const STATIC_JUMP = 1 << 0;
572        /// The jump target is known to be invalid.
573        /// Always returns [`InstructionResult::InvalidJump`] at runtime.
574        const INVALID_JUMP = 1 << 1;
575
576        /// The instruction is disabled in this EVM version.
577        /// Always returns [`InstructionResult::NotActivated`] at runtime.
578        const DISABLED = 1 << 3;
579        /// The instruction is unknown.
580        /// Always returns [`InstructionResult::NotFound`] at runtime.
581        const UNKNOWN = 1 << 4;
582
583        /// Skip generating instruction logic, but keep the gas calculation.
584        const SKIP_LOGIC = 1 << 6;
585        /// Don't generate any code.
586        const DEAD_CODE = 1 << 7;
587    }
588}

590fn bitvec_as_bytes<T: bitvec::store::BitStore, O: bitvec::order::BitOrder>(
591    bitvec: &BitVec<T, O>,
592) -> &[u8] {
593    slice_as_bytes(bitvec.as_raw_slice())
594}

596fn slice_as_bytes<T>(a: &[T]) -> &[u8] {
597    unsafe { std::slice::from_raw_parts(a.as_ptr().cast(), std::mem::size_of_val(a)) }
598}
