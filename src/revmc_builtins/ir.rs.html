4const MANGLE_PREFIX: &str = "__revmc_builtin_";

6/// Builtin cache.
7#[derive(Debug)]
8pub struct Builtins<B: Backend>([Option<B::Function>; Builtin::COUNT]);
9
10impl<B: Backend> Default for Builtins<B> {
11    fn default() -> Self {
12        Self::new()
13    }
14}
15
16impl<B: Backend> Builtins<B> {
17    /// Create a new cache.
18    pub fn new() -> Self {
19        Self([None; Builtin::COUNT])
20    }
21
22    /// Clear the cache.
23    pub fn clear(&mut self) {
24        *self = Self::new();
25    }
26
27    /// Get the function for the given builtin.
28    pub fn get(&mut self, builtin: Builtin, bcx: &mut B::Builder<'_>) -> B::Function {
29        *self.0[builtin as usize].get_or_insert_with(|| Self::init(builtin, bcx))
30    }
31
32    #[cold]
33    fn init(builtin: Builtin, bcx: &mut B::Builder<'_>) -> B::Function {
34        let name = builtin.name();
35        debug_assert!(name.starts_with(MANGLE_PREFIX), "{name:?}");
36        bcx.get_function(name).inspect(|r| trace!(name, ?r, "pre-existing")).unwrap_or_else(|| {
37            let r = Self::build(name, builtin, bcx);
38            trace!(name, ?r, "built");
39            r
40        })
41    }
42
43    fn build(name: &str, builtin: Builtin, bcx: &mut B::Builder<'_>) -> B::Function {
44        let ret = builtin.ret(bcx);
45        let params = builtin.params(bcx);
46        let address = builtin.addr();
47        let linkage = revmc_backend::Linkage::Import;
48        let f = bcx.add_function(name, &params, ret, Some(address), linkage);
49        let default_attrs: &[Attribute] = if builtin == Builtin::Panic {
50            &[
51                Attribute::Cold,
52                Attribute::NoReturn,
53                Attribute::NoFree,
54                Attribute::NoRecurse,
55                Attribute::NoSync,
56            ]
57        } else {
58            &[
59                Attribute::WillReturn,
60                Attribute::NoFree,
61                Attribute::NoRecurse,
62                Attribute::NoSync,
63                Attribute::NoUnwind,
64            ]
65        };
66        for attr in default_attrs.iter().chain(builtin.attrs()).copied() {
67            bcx.add_function_attribute(Some(f), attr, FunctionAttributeLocation::Function);
68        }
69        let param_attrs = builtin.param_attrs();
70        for (i, param_attrs) in param_attrs.iter().enumerate() {
71            for attr in param_attrs {
72                bcx.add_function_attribute(
73                    Some(f),
74                    *attr,
75                    FunctionAttributeLocation::Param(i as u32),
76                );
77            }
78        }
79        f
80    }
81}

83macro_rules! builtins {
84    (@count) => { 0 };
85    (@count $first:tt $(, $rest:tt)*) => { 1 + builtins!(@count $($rest),*) };
86
87    (@param_attr $default:ident) => { $default() };
88    (@param_attr $default:ident $name:expr) => { $name };
89
90    (@types |$bcx:ident| { $($types_init:tt)* }
91     @param_attrs |$op:ident| { $($attrs_init:tt)* }
92     $($ident:ident = $(#[$attr:expr])* $name:ident($($(@[$param_attr:expr])? $params:expr),* $(,)?) $ret:expr),* $(,)?
93    ) => { paste::paste! {
94        /// Builtins that can be called by the compiled functions.
95        #[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
96        pub enum Builtin {
97            $($ident,)*
98        }
99
100        #[allow(unused_variables)]
101        impl Builtin {
102            pub const COUNT: usize = builtins!(@count $($ident),*);
103
104            pub const fn name(self) -> &'static str {
105                match self {
106                    $(Self::$ident => stringify!($name),)*
107                }
108            }
109
110            pub fn addr(self) -> usize {
111                match self {
112                    $(Self::$ident => crate::$name as *const () as usize,)*
113                }
114            }
115
116            pub fn ret<B: TypeMethods>(self, $bcx: &mut B) -> Option<B::Type> {
117                $($types_init)*
118                match self {
119                    $(Self::$ident => $ret,)*
120                }
121            }
122
123            pub fn params<B: TypeMethods>(self, $bcx: &mut B) -> Vec<B::Type> {
124                $($types_init)*
125                match self {
126                    $(Self::$ident => vec![$($params),*],)*
127                }
128            }
129
130            pub fn attrs(self) -> &'static [Attribute] {
131                #[allow(unused_imports)]
132                use Attribute::*;
133                match self {
134                    $(Self::$ident => &[$($attr)*]),*
135                }
136            }
137
138            #[allow(non_upper_case_globals)]
139            pub fn param_attrs(self) -> Vec<Vec<Attribute>> {
140                #[allow(unused_imports)]
141                use Attribute::*;
142                let $op = self;
143                let default = || vec![Attribute::NoUndef];
144                $($attrs_init)*
145                match self {
146                    $(Self::$ident => vec![$(builtins!(@param_attr default $($param_attr)?)),*]),*
147                }
148            }
149
150            fn op(self) -> u8 {
151                use revm_bytecode::opcode::*;
152                const PANIC: u8 = 0;
153                const LOG: u8 = LOG0;
154                const DORETURN: u8 = RETURN;
155                const RESIZEMEMORY: u8 = 0;
156                const UDIV: u8 = DIV;
157                const UREM: u8 = MOD;
158
159                match self {
160                    $(Self::$ident => [<$ident:upper>]),*
161                }
162            }
163        }
164    }};
165}

167builtins! {
168    @types |bcx| {
169        let ptr = bcx.type_ptr();
170        let usize = bcx.type_ptr_sized_int();
171        let bool = bcx.type_int(1);
172        let u8 = bcx.type_int(8);
173    }
174
175    @param_attrs |op| {
176        fn size_and_align<T>() -> Vec<Attribute> {
177            size_and_align2(Some(core::mem::size_of::<T>()), core::mem::align_of::<T>())
178        }
179
180        fn size_and_align2(size: Option<usize>, align: usize) -> Vec<Attribute> {
181            let mut vec = Vec::with_capacity(5);
182            vec.push(Attribute::NoAlias);
183            vec.push(Attribute::NoCapture);
184            vec.push(Attribute::NoUndef);
185            vec.push(Attribute::Align(align as u64));
186            if let Some(size) = size {
187                vec.push(Attribute::Dereferenceable(size as u64));
188            }
189            vec
190        }
191
192        let op = op.op();
193        let (inputs, outputs) = if let Some(info) = revm_bytecode::opcode::OPCODE_INFO[op as usize] {
194            (info.inputs(), info.outputs())
195        } else {
196            (0, 0)
197        };
198
199        let ecx = size_and_align::<revmc_context::EvmContext<'static>>();
200
201        let sp_dyn = size_and_align2(None, core::mem::align_of::<revmc_context::EvmWord>());
202
203        let mut sp = sp_dyn.clone();
204        // `sp` is at `top - inputs`, we have access to `max(inputs, outputs)` words.
205        let n_stack_words = inputs.max(outputs);
206        let size_of_word = core::mem::size_of::<revmc_context::EvmWord>();
207        sp.push(Attribute::Dereferenceable(size_of_word as u64 * n_stack_words as u64));
208        match (inputs, outputs) {
209            (0, 0) => sp.push(Attribute::ReadNone),
210            (0, 1..) => sp.push(Attribute::WriteOnly),
211            (1.., 0) => sp.push(Attribute::ReadOnly),
212            (1.., 1..) => {}
213        }
214    }
