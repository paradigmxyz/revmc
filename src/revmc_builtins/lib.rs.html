36/// The kind of a `*CALL*` instruction.
37#[derive(Clone, Copy, Debug, PartialEq, Eq)]
38#[repr(u8)]
39pub enum CallKind {
40    /// `CALL`.
41    Call,
42    /// `CALLCODE`.
43    CallCode,
44    /// `DELEGATECALL`.
45    DelegateCall,
46    /// `STATICCALL`.
47    StaticCall,
48}
49
50impl From<CallKind> for CallScheme {
51    fn from(kind: CallKind) -> Self {
52        match kind {
53            CallKind::Call => Self::Call,
54            CallKind::CallCode => Self::CallCode,
55            CallKind::DelegateCall => Self::DelegateCall,
56            CallKind::StaticCall => Self::StaticCall,
57        }
58    }
59}

136pub unsafe extern "C" fn __revmc_builtin_balance(
137    ecx: &mut EvmContext<'_>,
138    address: &mut EvmWord,
139    spec_id: SpecId,
140) -> InstructionResult {
141    let state = try_host!(ecx.host.balance(address.to_address()));
142    *address = state.data.into();
143    let gas = if spec_id.is_enabled_in(SpecId::BERLIN) {
144        gas::warm_cold_cost(state.is_cold)
145    } else if spec_id.is_enabled_in(SpecId::ISTANBUL) {
146        // EIP-1884: Repricing for trie-size-dependent opcodes
147        700
148    } else if spec_id.is_enabled_in(SpecId::TANGERINE) {
149        400
150    } else {
151        20
152    };
153    gas!(ecx, gas);
154    InstructionResult::Stop
155}

166#[no_mangle]
167pub unsafe extern "C" fn __revmc_builtin_calldataload(
168    ecx: &mut EvmContext<'_>,
169    offset: &mut EvmWord,
170) {
171    let offset_usize = as_usize_saturated!(offset.to_u256());
172    let mut word = [0u8; 32];
173
174    match ecx.input.input() {
175        CallInput::Bytes(bytes) => {
176            let len = bytes.len().saturating_sub(offset_usize).min(32);
177            if len > 0 && offset_usize < bytes.len() {
178                word[..len].copy_from_slice(&bytes[offset_usize..offset_usize + len]);
179            }
180        }
181        CallInput::SharedBuffer(range) => {
182            let input_slice = ecx.memory.global_slice(range.clone());
183            let input_len = input_slice.len();
184            if offset_usize < input_len {
185                let count = 32.min(input_len - offset_usize);
186                word[..count].copy_from_slice(&input_slice[offset_usize..offset_usize + count]);
187            }
188        }
189    }
190
191    *offset = EvmWord::from_be_bytes(word);
192}

202#[no_mangle]
203pub unsafe extern "C" fn __revmc_builtin_calldatacopy(
204    ecx: &mut EvmContext<'_>,
205    rev![memory_offset, data_offset, len]: &mut [EvmWord; 3],
206) -> InstructionResult {
207    let len = try_into_usize!(len);
208    if len != 0 {
209        gas_opt!(ecx, gas::dyn_verylowcopy_cost(len as u64));
210        let memory_offset = try_into_usize!(memory_offset);
211        ensure_memory!(ecx, memory_offset, len);
212        let data_offset = as_usize_saturated!(data_offset.to_u256());
213
214        match ecx.input.input() {
215            CallInput::Bytes(bytes) => {
216                ecx.memory.set_data(memory_offset, data_offset, len, bytes.as_ref());
217            }
218            CallInput::SharedBuffer(range) => {
219                ecx.memory.set_data_from_global(memory_offset, data_offset, len, range.clone());
220            }
221        }
222    }
223    InstructionResult::Stop
224}

279#[no_mangle]
280pub unsafe extern "C" fn __revmc_builtin_returndatacopy(
281    ecx: &mut EvmContext<'_>,
282    rev![memory_offset, offset, len]: &mut [EvmWord; 3],
283) -> InstructionResult {
284    let len = try_into_usize!(len);
285    gas_opt!(ecx, gas::dyn_verylowcopy_cost(len as u64));
286    let data_offset = offset.to_u256();
287    let data_offset = as_usize_saturated!(data_offset);
288    let (data_end, overflow) = data_offset.overflowing_add(len);
289    if overflow || data_end > ecx.return_data.len() {
290        return InstructionResult::OutOfOffset;
291    }
292    if len != 0 {
293        let memory_offset = try_into_usize!(memory_offset);
294        ensure_memory!(ecx, memory_offset, len);
295        ecx.memory.set(memory_offset, &ecx.return_data[data_offset..data_end]);
296    }
297    InstructionResult::Stop
298}

319#[no_mangle]
320pub unsafe extern "C" fn __revmc_builtin_blockhash(
321    ecx: &mut EvmContext<'_>,
322    number_ptr: &mut EvmWord,
323) -> InstructionResult {
324    let requested_number = number_ptr.to_u256();
325    let block_number = ecx.host.block_number();
326
327    // Check if requested block is in the future
328    let Some(diff) = block_number.checked_sub(requested_number) else {
329        *number_ptr = EvmWord::ZERO;
330        return InstructionResult::Stop;
331    };
332
333    let diff = as_u64_saturated!(diff);
334
335    // Current block returns 0
336    if diff == 0 {
337        *number_ptr = EvmWord::ZERO;
338        return InstructionResult::Stop;
339    }
340
341    // BLOCK_HASH_HISTORY is 256
342    const BLOCK_HASH_HISTORY: u64 = 256;
343
344    if diff <= BLOCK_HASH_HISTORY {
345        let hash = try_host!(ecx.host.block_hash(as_u64_saturated!(requested_number)));
346        *number_ptr = EvmWord::from_be_bytes(hash.0);
347    } else {
348        // Too old, return 0
349        *number_ptr = EvmWord::ZERO;
350    }
351
352    InstructionResult::Stop
353}

495#[no_mangle]
496pub unsafe extern "C" fn __revmc_builtin_log(
497    ecx: &mut EvmContext<'_>,
498    sp: *mut EvmWord,
499    n: u8,
500) -> InstructionResult {
501    ensure_non_staticcall!(ecx);
502    assume!(n <= 4, "invalid log topic count: {n}");
503    let sp = sp.add(n as usize);
504    read_words!(sp, offset, len);
505    let len = try_into_usize!(len);
506    gas_opt!(ecx, gas::dyn_log_cost(len as u64));
507    let data = if len != 0 {
508        let offset = try_into_usize!(offset);
509        ensure_memory!(ecx, offset, len);
510        Bytes::copy_from_slice(&ecx.memory.slice(offset..offset + len))
511    } else {
512        Bytes::new()
513    };
514
515    let mut topics = Vec::with_capacity(n as usize);
516    for i in 1..=n {
517        topics.push(sp.sub(i as usize).read().to_be_bytes().into());
518    }
519
520    ecx.host.log(Log {
521        address: ecx.input.target_address,
522        data: LogData::new(topics, data).expect("too many topics"),
523    });
524    InstructionResult::Stop
525}

530#[no_mangle]
531pub unsafe extern "C" fn __revmc_builtin_create(
532    ecx: &mut EvmContext<'_>,
533    sp: *mut EvmWord,
534    spec_id: SpecId,
535    create_kind: CreateKind,
536) -> InstructionResult {
537    ensure_non_staticcall!(ecx);
538
539    let len = match create_kind {
540        CreateKind::Create => 3,
541        CreateKind::Create2 => 4,
542    };
543    let mut sp = sp.add(len);
544    pop!(sp; value, code_offset, len);
545
546    let len = try_into_usize!(len);
547    let code = if len != 0 {
548        if spec_id.is_enabled_in(SpecId::SHANGHAI) {
549            // Limit is set as double of max contract bytecode size
550            let max_initcode_size = ecx.host.max_initcode_size();
551            if len > max_initcode_size {
552                return InstructionResult::CreateInitCodeSizeLimit;
553            }
554            gas!(ecx, gas::initcode_cost(len as u64));
555        }
556
557        let code_offset = try_into_usize!(code_offset);
558        ensure_memory!(ecx, code_offset, len);
559        Bytes::copy_from_slice(&ecx.memory.slice(code_offset..code_offset + len))
560    } else {
561        Bytes::new()
562    };
563
564    let is_create2 = create_kind == CreateKind::Create2;
565    gas_opt!(ecx, if is_create2 { gas::create2_cost(len as u64) } else { Some(gas::CREATE) });
566
567    let scheme = if is_create2 {
568        pop!(sp; salt);
569        CreateScheme::Create2 { salt: salt.to_u256() }
570    } else {
571        CreateScheme::Create
572    };
573
574    let mut gas_limit = ecx.gas.remaining();
575    if spec_id.is_enabled_in(SpecId::TANGERINE) {
576        gas_limit -= gas_limit / 64;
577    }
578    gas!(ecx, gas_limit);
579
580    *ecx.next_action =
581        Some(InterpreterAction::NewFrame(revm_interpreter::FrameInput::Create(Box::new(
582            CreateInputs::new(ecx.input.target_address, scheme, value.to_u256(), code, gas_limit),
583        ))));
584
585    InstructionResult::Stop
586}

588#[no_mangle]
589pub unsafe extern "C" fn __revmc_builtin_call(
590    ecx: &mut EvmContext<'_>,
591    sp: *mut EvmWord,
592    spec_id: SpecId,
593    call_kind: CallKind,
594) -> InstructionResult {
595    let len = match call_kind {
596        CallKind::Call | CallKind::CallCode => 7,
597        CallKind::DelegateCall | CallKind::StaticCall => 6,
598    };
599    let mut sp = sp.add(len);
600
601    pop!(sp; local_gas_limit, to);
602    let local_gas_limit = local_gas_limit.to_u256();
603    let to = to.to_address();
604
605    // max gas limit is not possible in real ethereum situation.
606    // But for tests we would not like to fail on this.
607    // Gas limit for subcall is taken as min of this value and current gas limit.
608    let local_gas_limit = as_u64_saturated!(local_gas_limit);
609
610    let value = match call_kind {
611        CallKind::Call | CallKind::CallCode => {
612            pop!(sp; value);
613            let value = value.to_u256();
614            if call_kind == CallKind::Call && ecx.is_static && value != U256::ZERO {
615                return InstructionResult::CallNotAllowedInsideStatic;
616            }
617            value
618        }
619        CallKind::DelegateCall | CallKind::StaticCall => U256::ZERO,
620    };
621    let transfers_value = value != U256::ZERO;
622
623    pop!(sp; in_offset, in_len, out_offset, out_len);
624
625    let in_len = try_into_usize!(in_len);
626    let input = if in_len != 0 {
627        let in_offset = try_into_usize!(in_offset);
628        ensure_memory!(ecx, in_offset, in_len);
629        Bytes::copy_from_slice(&ecx.memory.slice(in_offset..in_offset + in_len))
630    } else {
631        Bytes::new()
632    };
633
634    let out_len = try_into_usize!(out_len);
635    let out_offset = if out_len != 0 {
636        let out_offset = try_into_usize!(out_offset);
637        ensure_memory!(ecx, out_offset, out_len);
638        out_offset
639    } else {
640        usize::MAX // unrealistic value so we are sure it is not used
641    };
642
643    // Charge the CALL base access cost up-front. In the interpreter this is charged as static
644    // opcode gas before entering call helpers; revmc marks CALL as dynamic, so the builtin must
645    // do it.
646    gas!(ecx, ecx.host.gas_params().warm_storage_read_cost());
647
648    if transfers_value {
649        gas!(ecx, ecx.host.gas_params().transfer_value_cost());
650    }
651
652    // Match interpreter call path: load delegated account and pass resolved bytecode/hash
653    // through CallInputs::known_bytecode (covers EIP-7702 delegation and EOF execution).
654    let (dynamic_gas, bytecode, code_hash) =
655        match revm_interpreter::instructions::contract::load_account_delegated(
656            ecx.host,
657            spec_id,
658            ecx.gas.remaining(),
659            to,
660            transfers_value,
661            call_kind == CallKind::Call,
662        ) {
663            Ok(out) => out,
664            Err(revm_context_interface::host::LoadError::ColdLoadSkipped) => {
665                return InstructionResult::OutOfGas;
666            }
667            Err(revm_context_interface::host::LoadError::DBError) => {
668                return InstructionResult::FatalExternalError;
669            }
670        };
671
672    gas!(ecx, dynamic_gas);
673
674    // EIP-150: Gas cost changes for IO-heavy operations
675    let mut gas_limit = if spec_id.is_enabled_in(SpecId::TANGERINE) {
676        let gas = ecx.gas.remaining();
677        // take l64 part of gas_limit
678        (gas - gas / 64).min(local_gas_limit)
679    } else {
680        local_gas_limit
681    };
682
683    gas!(ecx, gas_limit);
684
685    // Add call stipend if there is value to be transferred.
686    if matches!(call_kind, CallKind::Call | CallKind::CallCode) && transfers_value {
687        gas_limit = gas_limit.saturating_add(ecx.host.gas_params().call_stipend());
688    }
689
690    *ecx.next_action = Some(InterpreterAction::NewFrame(revm_interpreter::FrameInput::Call(
691        Box::new(CallInputs {
692            input: CallInput::Bytes(input),
693            return_memory_offset: out_offset..out_offset + out_len,
694            gas_limit,
695            bytecode_address: to,
696            known_bytecode: Some((code_hash, bytecode)),
697            target_address: if matches!(call_kind, CallKind::DelegateCall | CallKind::CallCode) {
698                ecx.input.target_address
699            } else {
700                to
701            },
702            caller: if call_kind == CallKind::DelegateCall {
703                ecx.input.caller_address
704            } else {
705                ecx.input.target_address
706            },
707            value: if call_kind == CallKind::DelegateCall {
708                CallValue::Apparent(ecx.input.call_value)
709            } else {
710                CallValue::Transfer(value)
711            },
712            scheme: call_kind.into(),
713            is_static: ecx.is_static || call_kind == CallKind::StaticCall,
714        }),
715    )));
716
717    InstructionResult::Stop
718}

720#[no_mangle]
721pub unsafe extern "C" fn __revmc_builtin_do_return(
722    ecx: &mut EvmContext<'_>,
723    rev![offset, len]: &mut [EvmWord; 2],
724    result: InstructionResult,
725) -> InstructionResult {
726    let len = try_into_usize!(len);
727    let output = if len != 0 {
728        let offset = try_into_usize!(offset);
729        ensure_memory!(ecx, offset, len);
730        ecx.memory.slice(offset..offset + len).to_vec().into()
731    } else {
732        Bytes::new()
733    };
734    *ecx.next_action =
735        Some(InterpreterAction::Return(InterpreterResult { output, gas: *ecx.gas, result }));
736    InstructionResult::Stop
737}

739#[no_mangle]
740pub unsafe extern "C" fn __revmc_builtin_selfdestruct(
741    ecx: &mut EvmContext<'_>,
742    target: &mut EvmWord,
743    spec_id: SpecId,
744) -> InstructionResult {
745    ensure_non_staticcall!(ecx);
746
747    // EIP-150: SELFDESTRUCT base cost is 5000 starting from TANGERINE
748    if spec_id.is_enabled_in(SpecId::TANGERINE) {
749        gas!(ecx, 5000);
750    }
751
752    let res = match ecx.host.selfdestruct(ecx.input.target_address, target.to_address(), false) {
753        Ok(r) => r,
754        Err(_) => return InstructionResult::FatalExternalError,
755    };
756
757    // EIP-161: State trie clearing (invariant-preserving alternative)
758    let should_charge_topup = if spec_id.is_enabled_in(SpecId::SPURIOUS_DRAGON) {
759        res.data.had_value && !res.data.target_exists
760    } else {
761        !res.data.target_exists
762    };
763
764    gas!(ecx, ecx.host.gas_params().selfdestruct_cost(should_charge_topup, res.is_cold));
765
766    if !res.data.previously_destroyed {
767        ecx.gas.record_refund(ecx.host.gas_params().selfdestruct_refund());
768    }
769
770    InstructionResult::SelfDestruct
771}

782pub unsafe extern "C" fn __revmc_builtin_mload(
783    ecx: &mut EvmContext<'_>,
784    rev![offset_ptr]: &mut [EvmWord; 1],
785) -> InstructionResult {
786    let offset = try_into_usize!(offset_ptr);
787    ensure_memory!(ecx, offset, 32);
788    let slice = ecx.memory.slice(offset..offset + 32);
789    let mut word = [0u8; 32];
790    word.copy_from_slice(&slice);
791    *offset_ptr = EvmWord::from_be_bytes(word);
792    InstructionResult::Stop
793}
